SELECT I.*, L.NAME
FROM ISSUESS I, ISSUESLISTS L

SELECT I.*, U.FIRSTNAME, U.LASTNAME 
FROM ISSUESS I, USERS U

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS TASK_NUMER
FROM ISSUESS I, USERS U
WHERE I.USER_ID_ASSIGNEDTO = U.ID
GROUP BY I.USER_ID_ASSIGNEDTO
HAVING COUNT(*) > 0
ORDER BY U.LASTNAME, U.FIRSTNAME;
